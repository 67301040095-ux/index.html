// ---------------------- // PIPE FUNCTION // ---------------------- function pipeVertical(x,z,y1,y2,color){ const length = Math.abs(y2-y1); const pipe = new THREE.Mesh( new THREE.CylinderGeometry(0.15,0.15,length,16), new THREE.MeshStandardMaterial({color:color}) ); pipe.position.set(x,(y1+y2)/2,z); scene.add(pipe); } function pipeHorizontal(x1,x2,y,z,color){ const length = Math.abs(x2-x1); const pipe = new THREE.Mesh( new THREE.CylinderGeometry(0.15,0.15,length,16), new THREE.MeshStandardMaterial({color:color}) ); pipe.rotation.z = Math.PI/2; pipe.position.set((x1+x2)/2,y,z); scene.add(pipe); } // ---------------------- // HOT WATER PIPE (แดง) // CT → CH → AHU // ---------------------- pipeVertical(-1,10,2,1,0xff0000); // CT ลง pipeVertical(-1,1,2,1,0xff0000); // CH ลง pipeVertical(-1,-8,2,1,0xff0000); // AHU ลง // ---------------------- // COLD WATER PIPE (ฟ้า) // AHU → CH → CT // ---------------------- pipeVertical(1,-10,1,2,0x00ffff); // AHU ขึ้น pipeVertical(1,-1,1,2,0x00ffff); // CH ขึ้น pipeVertical(1,8,1,2,0x00ffff); // CT ขึ้น
